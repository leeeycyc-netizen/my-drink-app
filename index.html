    <script src="https://unpkg.com/react@18/umd/react.development.js"></script>

    <script src="https://unpkg.com/react-dom@18/umd/react-dom.development.js"></script>

    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>

    <!-- 引入 Tailwind CSS -->

    <script src="https://cdn.tailwindcss.com"></script>

    <!-- 引入 Lucide Icons -->

    <script src="https://unpkg.com/lucide@latest"></script>

    <script src="https://unpkg.com/lucide-react@latest"></script>

    <style>

        @import url('https://fonts.googleapis.com/css2?family=Noto+Sans+TC:wght@400;500;700;900&display=swap');

        body { font-family: 'Noto Sans TC', sans-serif; }

        input[type="range"]::-webkit-slider-thumb {

<body>

    <div id="root"></div>


    <script type="text/babel">

        const { useState, useEffect, useMemo } = React;

        const { 

            Plus, History, BarChart3, Coffee, Star, Trash2, X, 

            TrendingUp, Heart, Calendar, Check, Award, 

            ChevronRight, Smartphone, Share, ShoppingBag, ListChecks 

        } = lucideReact;


        const STORAGE_KEY = 'drink_log_2026_pink_v4';

        const SWEETNESS_OPTIONS = ['無糖', '1分', '微糖', '半糖', '少糖', '正常'];

        const ICE_OPTIONS = ['去冰', '微冰', '少冰', '正常', '熱飲'];


        const App = () => {

            const [drinks, setDrinks] = useState([]);

            const [isModalOpen, setIsModalOpen] = useState(false);

            const [isGuideOpen, setIsGuideOpen] = useState(false);

            const [activeTab, setActiveTab] = useState('history');

            const [formData, setFormData] = useState({

                shop: '', item: '', sweetness: '半糖', ice: '少冰',

                price: '', rating: 5, comment: '',

                date: new Date().toISOString().split('T')[0]

            });


            useEffect(() => {

                const saved = localStorage.getItem(STORAGE_KEY);

                if (saved) {

                    try { setDrinks(JSON.parse(saved)); } catch (e) { console.error(e); }

                }

            }, []);


            useEffect(() => {

                localStorage.setItem(STORAGE_KEY, JSON.stringify(drinks));

            }, [drinks]);


            const stats = useMemo(() => {

                const totalCount = drinks.length;

                const totalPrice = drinks.reduce((sum, d) => sum + (Number(d.price) || 0), 0);

                const avgPrice = totalCount > 0 ? Math.round(totalPrice / totalCount) : 0;

                const avgRating = totalCount > 0 ? (drinks.reduce((sum, d) => sum + d.rating, 0) / totalCount).toFixed(1) : 0;

                return { totalCount, totalPrice, avgPrice, avgRating };

            }, [drinks]);


            const handleInputChange = (e) => {

                const { name, value } = e.target;

                setFormData(prev => ({ ...prev, [name]: value }));

            };


            const handleAddDrink = (e) => {

                e.preventDefault();

                if (!formData.shop || !formData.item || !formData.price) return;

                const newDrink = { ...formData, id: Date.now() };

                setDrinks([newDrink, ...drinks]);

                setIsModalOpen(false);

                setFormData({

                    shop: '', item: '', sweetness: '半糖', ice: '少冰',

                    price: '', rating: 5, comment: '',

                    date: new Date().toISOString().split('T')[0]

                });

            };


            const deleteDrink = (id) => {

                setDrinks(drinks.filter(d => d.id !== id));

            };


            return (

                <div className="min-h-screen bg-[#FFF5F7] text-slate-900 pb-28">

                    <header className="sticky top-0 z-30 bg-white/80 backdrop-blur-md px-6 pt-12 pb-6 border-b border-pink-50">

                        <div className="flex justify-between items-center max-w-2xl mx-auto">

                            <div>

                                <h1 className="text-2xl font-black text-pink-600 flex items-center gap-2">

                                    超愛喝飲料 <Heart className="fill-pink-500 text-pink-500" size={20} />

                                </h1>

                                <p className="text-pink-400 text-[10px] font-black uppercase tracking-widest">Beverage Lover</p>

                            </div>

                            <button onClick={() => setIsGuideOpen(true)} className="w-10 h-10 bg-pink-50 text-pink-500 rounded-xl flex items-center justify-center"><Smartphone size={20} /></button>

                        </div>

                    </header>


                    <div className="max-w-2xl mx-auto px-6 mt-6">

                        <section className="grid grid-cols-2 gap-4 mb-8">

                            <div className="bg-white p-5 rounded-[2rem] shadow-sm border border-pink-50">

                                <p className="text-pink-300 text-[10px] font-black uppercase tracking-widest mb-1">Total Spent</p>

                                <div className="text-3xl font-black text-pink-500 tracking-tighter">${stats.totalPrice}</div>

                                <div className="mt-3 flex items-center gap-1 text-[10px] font-bold text-pink-400"><ShoppingBag size={12} /><span>支出明細</span></div>

                            </div>

                            <div className="bg-white p-5 rounded-[2rem] shadow-sm border border-pink-50">

                                <p className="text-amber-300 text-[10px] font-black uppercase tracking-widest mb-1">Cup Count</p>

                                <div className="text-3xl font-black text-amber-500 tracking-tighter">{stats.totalCount} <span className="text-xs">杯</span></div>

                                <div className="mt-3 flex items-center gap-1 text-[10px] font-bold text-amber-500"><ListChecks size={12} /><span>飲用統計</span></div>

                            </div>

                        </section>


                        {activeTab === 'history' ? (

                            <div className="space-y-4">

                                {drinks.length === 0 ? (

                                    <div className="bg-white rounded-[2.5rem] p-12 text-center border-2 border-dashed border-pink-100 text-pink-300 font-bold">還沒有紀錄喔！點下方的 + 開始</div>

                                ) : (

                                    drinks.map(drink => (

                                        <div key={drink.id} className="bg-white p-5 rounded-[2.5rem] border border-pink-50">

                                            <div className="flex justify-between items-start">

                                                <div className="flex gap-4">

                                                    <div className="w-14 h-14 bg-pink-50 rounded-3xl flex items-center justify-center text-pink-400"><Coffee size={24} /></div>

                                                    <div>

                                                        <h3 className="font-black text-slate-800 text-lg">{drink.shop}</h3>

                                                        <p className="text-slate-500 text-sm">{drink.item}</p>

                                                        <div className="flex gap-2 mt-2">

                                                            <span className="text-[10px] font-bold px-3 py-1 bg-pink-50 rounded-full text-pink-500">{drink.sweetness}</span>

                                                            <span className="text-[10px] font-bold px-3 py-1 bg-pink-50 rounded-full text-pink-500">{drink.ice}</span>

                                                        </div>

                                                    </div>

                                                </div>

                                                <div className="text-right">

                                                    <p className="font-black text-xl text-pink-500">${drink.price}</p>

                                                    <div className="flex gap-0.5 mt-1 justify-end">

                                                        {[1,2,3,4,5].map(s => <Star key={s} size={10} className={s <= drink.rating ? 'fill-pink-400 text-pink-400' : 'text-pink-100'} />)}

                                                    </div>

                                                </div>

                                            </div>

                                            {drink.comment && <div className="mt-3 bg-pink-50/50 p-3 rounded-2xl text-xs text-pink-600 border-l-4 border-pink-300">“ {drink.comment} ”</div>}

                                            <div className="flex justify-between items-center pt-3 mt-3 border-t border-pink-50">

                                                <span className="text-[10px] font-bold text-pink-300 uppercase flex items-center gap-1"><Calendar size={12} /> {drink.date}</span>

                                                <button onClick={() => deleteDrink(drink.id)} className="text-pink-200 hover:text-rose-500 transition-all"><Trash2 size={16} /></button>

                                            </div>

                                        </div>

                                    ))

                                )}

                            </div>

                        ) : (

                            <div className="animate-in fade-in slide-in-from-bottom-4 duration-500">

                                <div className="bg-pink-500 rounded-[3rem] p-8 text-white shadow-2xl shadow-pink-100 relative overflow-hidden mb-6">

                                    <div className="absolute -right-10 -bottom-10 w-40 h-40 bg-pink-400/50 rounded-full blur-3xl"></div>

                                    <div className="relative z-10">

                                        <p className="text-pink-100 text-sm font-bold mb-1 uppercase tracking-widest">Yearly Report</p>

                                        <div className="text-5xl font-black tracking-tighter mb-2">${stats.totalPrice}</div>

                                        <p className="text-pink-100 text-[10px] font-bold uppercase tracking-widest">Average ${stats.avgPrice} per cup</p>

                                    </div>

                                </div>

                                <div className="grid grid-cols-2 gap-4">

                                    <div className="bg-white p-6 rounded-[2.5rem] border border-pink-50 text-center">

                                        <p className="text-pink-300 text-[10px] font-black uppercase tracking-widest mb-1">平均評分</p>

                                        <div className="flex items-center justify-center gap-2"><span className="text-3xl font-black text-pink-500">{stats.avgRating}</span><Star size={20} className="fill-pink-400 text-pink-400" /></div>

                                    </div>

                                    <div className="bg-white p-6 rounded-[2.5rem] border border-pink-50">

                                        <p className="text-pink-300 text-[10px] font-black uppercase tracking-widest mb-1">消費總量</p>

                                        <p className="text-sm font-black text-slate-800">{stats.totalCount} 次紀錄</p>

                                    </div>

                                </div>

                            </div>

                        )}

                    </div>


                    <nav className="fixed bottom-6 left-1/2 -translate-x-1/2 w-[90%] max-w-md bg-white/90 backdrop-blur-xl border border-pink-100 rounded-[2.5rem] px-8 py-3 flex justify-between items-center z-40 shadow-2xl">

                        <button onClick={() => setActiveTab('history')} className={`flex flex-col items-center gap-1 ${activeTab === 'history' ? 'text-pink-600' : 'text-pink-200'}`}>

                            <History size={24} /><span className="text-[10px] font-bold uppercase tracking-tighter">紀錄</span>

                        </button>

                        <button onClick={() => setIsModalOpen(true)} className="bg-pink-500 text-white w-14 h-14 rounded-2xl flex items-center justify-center -mt-12 shadow-xl border-4 border-white"><Plus size={32} /></button>

                        <button onClick={() => setActiveTab('stats')} className={`flex flex-col items-center gap-1 ${activeTab === 'stats' ? 'text-pink-600' : 'text-pink-200'}`}>

                            <BarChart3 size={24} /><span className="text-[10px] font-bold uppercase tracking-tighter">分析</span>

                        </button>

                    </nav>


                    {isGuideOpen && (

                        <div className="fixed inset-0 z-50 flex items-center justify-center p-6 bg-slate-900/40 backdrop-blur-sm">

                            <div className="bg-white w-full max-w-sm rounded-[3rem] p-8 animate-in zoom-in-95 duration-300">

                                <h2 className="text-xl font-black text-pink-600 mb-6 flex items-center gap-2"><Smartphone size={24} /> 設為手機捷徑</h2>

                                <div className="space-y-6">

                                    <div className="flex gap-4"><div className="w-8 h-8 bg-pink-50 rounded-full flex items-center justify-center text-pink-500 font-black shrink-0">1</div><p className="text-sm text-slate-600"><span className="font-bold text-slate-800">點擊分享選單</span><br/>iPhone 請點 Safari 下方的分享圖示。</p></div>

                                    <div className="flex gap-4"><div className="w-8 h-8 bg-pink-50 rounded-full flex items-center justify-center text-pink-500 font-black shrink-0">2</div><p className="text-sm text-slate-600"><span className="font-bold text-slate-800">加入主畫面</span><br/>找到「加入主畫面」並點擊新增。</p></div>

                                </div>

                                <button onClick={() => setIsGuideOpen(false)} className="w-full bg-pink-500 text-white font-black py-4 rounded-2xl mt-8">我知道了！</button>

                            </div>

                        </div>

                    )}


                    {isModalOpen && (

                        <div className="fixed inset-0 z-50 flex items-end justify-center bg-slate-900/40 backdrop-blur-sm">

                            <div className="bg-white w-full max-w-lg rounded-t-[3.5rem] p-8 max-h-[95vh] overflow-y-auto animate-in slide-in-from-bottom-full duration-500">

                                <div className="flex justify-between items-center mb-8">

                                    <h2 className="text-2xl font-black text-pink-600 tracking-tight">今天的飲料紀錄</h2>

                                    <button onClick={() => setIsModalOpen(false)} className="bg-pink-50 p-3 rounded-2xl text-pink-300"><X size={20} /></button>

                                </div>

                                <form onSubmit={handleAddDrink} className="space-y-6">

                                    <input name="shop" required placeholder="哪家飲料店？" className="w-full bg-pink-50/50 rounded-[1.5rem] px-6 py-4 font-bold outline-none border-2 border-transparent focus:border-pink-200" value={formData.shop} onChange={handleInputChange} />

                                    <input name="item" required placeholder="點了什麼品項？" className="w-full bg-pink-50/50 rounded-[1.5rem] px-6 py-4 font-bold outline-none border-2 border-transparent focus:border-pink-200" value={formData.item} onChange={handleInputChange} />

                                    <div className="space-y-2"><p className="text-[10px] font-black text-pink-300 uppercase px-2 tracking-widest">甜度</p>

                                        <div className="grid grid-cols-3 gap-2">{SWEETNESS_OPTIONS.map(opt => (

                                            <button key={opt} type="button" onClick={() => setFormData(p => ({...p, sweetness: opt}))} className={`py-3 rounded-2xl text-xs font-black transition-all ${formData.sweetness === opt ? 'bg-pink-500 text-white shadow-lg' : 'bg-pink-50/50 text-pink-300'}`}>{opt}</button>

                                        ))}</div>

                                    </div>

                                    <div className="space-y-2"><p className="text-[10px] font-black text-pink-300 uppercase px-2 tracking-widest">冰塊</p>

                                        <div className="flex flex-wrap gap-2">{ICE_OPTIONS.map(opt => (

                                            <button key={opt} type="button" onClick={() => setFormData(p => ({...p, ice: opt}))} className={`px-5 py-3 rounded-2xl text-xs font-black transition-all ${formData.ice === opt ? 'bg-pink-500 text-white shadow-lg' : 'bg-pink-50/50 text-pink-300'}`}>{opt}</button>

                                        ))}</div>

                                    </div>

                                    <div className="grid grid-cols-2 gap-4">

                                        <input name="price" type="number" required placeholder="價格 $" className="w-full bg-pink-50/50 rounded-2xl px-6 py-4 font-black outline-none border-2 border-transparent focus:border-pink-200" value={formData.price} onChange={handleInputChange} />

                                        <input name="date" type="date" required className="w-full bg-pink-50/50 rounded-2xl px-6 py-4 font-bold outline-none border-2 border-transparent focus:border-pink-200" value={formData.date} onChange={handleInputChange} />

                                    </div>

                                    <div className="space-y-4 pt-4">

                                        <div className="flex justify-between px-2 text-[10px] font-black text-pink-300 uppercase">評價 ({formData.rating}) <div className="flex gap-1">{[1,2,3,4,5].map(s => <Star key={s} size={14} className={s <= formData.rating ? 'fill-pink-400 text-pink-400' : 'text-pink-50'} />)}</div></div>

                                        <input type="range" min="1" max="5" step="1" className="w-full h-3 bg-pink-50 rounded-full appearance-none accent-pink-500" value={formData.rating} onChange={(e) => setFormData(prev => ({...prev, rating: parseInt(e.target.value)}))} />

                                    </div>

                                    <textarea name="comment" placeholder="這杯味道如何？" className="w-full bg-pink-50/50 rounded-[1.5rem] px-6 py-4 font-medium outline-none h-24 border-2 border-transparent focus:border-pink-200" value={formData.comment} onChange={handleInputChange}></textarea>

                                    <button type="submit" className="w-full bg-pink-500 text-white font-black py-5 rounded-[1.5rem] shadow-xl hover:bg-pink-600 active:scale-95 transition-all flex items-center justify-center gap-2"><Check size={20} /> 儲存日誌</button>

                                </form>

                            </div>

                        </div>

                    )}

                </div>

            );

        };


        const root = ReactDOM.createRoot(document.getElementById('root'));

        root.render(<App />);

    </script>
